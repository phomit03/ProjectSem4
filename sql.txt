CREATE TABLE team (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    logo_img VARCHAR(255),
    coach INT,
    home_stadium VARCHAR(255),
    club_valuation FLOAT,
    status INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE feedback (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id VARCHAR(255) NOT NULL,
    subject INT,
    content TEXT,
    status INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE player (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    avatar_img VARCHAR(255),
    date_of_birth INT,
    national VARCHAR(255),
    position VARCHAR(255),
    number INT,
    weight INT,
    height INT,
    achievement TEXT,
    team_id INT,
    status INT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

CREATE TABLE match (
    id INT AUTO_INCREMENT PRIMARY KEY,
    date DATE,
    time TIME,
    stadium VARCHAR(255),
    status INT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

CREATE TABLE match_detail (
    id INT AUTO_INCREMENT PRIMARY KEY,
    match_id INT,
    home_team_id INT,
    away_team_id INT,
    player_goal TEXT,
    shot TEXT,
    shot_on_target TEXT,
    possesion TEXT,
    foul TEXT,
    passes TEXT,
    pass_accuracy TEXT,
    red_card TEXT,
    yellow_card TEXT,
    off_side TEXT,
    corner TEXT,
    status INT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

CREATE TABLE news (
    id INT AUTO_INCREMENT PRIMARY KEY,
    image VARCHAR(255),
    title VARCHAR(255),
    content TEXT,
    status INT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

CREATE TABLE user (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255),
    email VARCHAR(255),
    password INT,
    address VARCHAR(255),
    date_of_birth DATE,
    full_name VARCHAR(255),
    phone VARCHAR(255),
    status INT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

CREATE TABLE role (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    status INT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

CREATE TABLE user_role (
    user_id INT,
    role_id INT,
    status INT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
